71 uint8_t rcProcRrmPolicyValue(int id, void *value, int size){
72  int *intValue = (int*)value;
73  char *charValue = (char*)value;
74  static int memberIdx = 0;
75 
76  // TODO: use linked-list
77 
78  if(size){
79  switch(id){
80  case POLICY_RATIO_LIST:
81  case POLICY_RATIO_GROUP:
82  case RRM_POLICY:
83  case POLICY_MEMBER_LIST:
84  case POLICY_MEMBER:
85  case POLICY_MEMBER_SNSSAI:
86  break;
87  case POLICY_MEMBER_PLMNID:
88  memcpy(rrmPolicyGroup[policyIdx].rRMPolicyMemberList[memberIdx].mcc, charValue, 3);
89  memcpy(rrmPolicyGroup[policyIdx].rRMPolicyMemberList[memberIdx].mnc, charValue+3, 3);
90  break;
91  case POLICY_MEMBER_SST:
92  rrmPolicyGroup[policyIdx].rRMPolicyMemberList[memberIdx].sst = *charValue;
93  break;
94  case POLICY_MEMBER_SD:
95  memcpy(rrmPolicyGroup[policyIdx].rRMPolicyMemberList[memberIdx].sd, charValue, 3);
96  memberIdx++;
97  break;
98  case POLICY_MIN_PRB_RATIO:
99  rrmPolicyGroup[policyIdx].rRMPolicyMinRatio = *intValue;
100  break;
101  case POLICY_MAX_PRB_RATIO:
102  rrmPolicyGroup[policyIdx].rRMPolicyMaxRatio = *intValue;
103  break;
104  case POLICY_DED_PRB_RATIO:
105  rrmPolicyGroup[policyIdx].rRMPolicyDedicatedRatio = *intValue;
106  rrmPolicyGroup[policyIdx].id[0] = policyIdx + 1;
107  rrmPolicyGroup[policyIdx].rRMMemberNum = memberIdx;
108  policyIdx++;
109  break;
110  default:
111  DU_LOG("\nERROR --> E2SM-RC : Invaild parameter ID, %d", id);
112  return RFAILED;
113  }
114  }
115  else{
116  switch(id){
117  case POLICY_RATIO_LIST:
118  policyIdx = 0;
119  memberIdx = 0;
120  rrmPolicyGroup[policyIdx].resourceType = PRB;
121  break;
122  case RRM_POLICY:
123  memberIdx = 0;
124  break;
125  case POLICY_MEMBER_LIST:
126  case POLICY_RATIO_GROUP:
127  case POLICY_MEMBER_PLMNID:
128  case POLICY_MEMBER:
129  case POLICY_MEMBER_SNSSAI:
130  case POLICY_MEMBER_SST:
131  case POLICY_MEMBER_SD:
132  case POLICY_MIN_PRB_RATIO:
133  case POLICY_MAX_PRB_RATIO:
134  case POLICY_DED_PRB_RATIO:
135  break;
136  default:
137  DU_LOG("\nERROR --> E2SM-RC : Invaild parameter ID, %d", id);
138  return RFAILED;
139  break;
140  }
141  }
142  return ROK;
143 }
