301 {

303  * PHY_DELTA - the physical layer delta *
304  * SCHED_DELTA - scheduler schedules one slot ahead *
305  * BO_DELTA - this delay is considered for BO response and *

307  * lower-mac (FAPI filling) will be working on PHY_DELTA *
308  * brdcast scheduler will working on PHY_DELTA + SCHED_DELTA *
309  * RAR scheduler will working on PHY_DELTA + SCHED_DELTA *

312  * on PHY_DELTA + SCHED_DELTA + BO_DELTA *

314 
315  ADD_DELTA_TO_TIME(slotInd, schSlotValue->currentTime, PHY_DELTA_DL, numOfSlots);
