139 /**************************************************************************
140  * @brief Task Activation callback function.
141  *
142  * @details
143  *
144  * Function : lwrMacActvTsk
145  *
146  * Functionality:
147  * Primitives invoked by Lower MAC's users/providers through
148  * a loosely coupled interface arrive here by means of
149  * SSI's message handling. This API is registered with
150  * SSI during the Task Registration of DU APP.
151  *
152  * @param[in] Pst *pst, Post structure of the primitive.
153  * @param[in] Buffer *mBuf, Packed primitive parameters in the
154  * buffer.
155  * @return ROK - success
156  * RFAILED - failure
157  *
158  ***************************************************************************/
159 uint8_t lwrMacActvTsk(Pst *pst, Buffer *mBuf)
160 {
161  uint8_t ret = ROK;
162 
163 #ifdef CALL_FLOW_DEBUG_LOG
164  callFlowlwrMacActvTsk(pst);
165 #endif
166 
167  switch(pst->srcEnt)
168  {
169  case ENTLWRMAC:
170  {
171  switch(pst->event)
172  {
173 #ifdef INTEL_WLS_MEM
174  case EVT_START_WLS_RCVR:
175  {
176  ODU_PUT_MSG_BUF(mBuf);
177 
178  /* Allocate memory for intial UL transmission */
179  LwrMacEnqueueWlsBlock();
180 
181  /* Start thread to receive from L1 */
182  DU_LOG("\nINFO --> LWR MAC: Starting WLS receiver thread");
183  LwrMacRecvPhyMsg();
184  break;
185  }
186 #endif
187  default:
188  {
189  ODU_PUT_MSG_BUF(mBuf);
190  DU_LOG("\nERROR --> LWR MAC: Invalid event %d received", pst->event);
191  ret = RFAILED;
192  }
193  }
194  break;
195  }
196 
197 #ifndef INTEL_WLS_MEM
198  case ENTPHYSTUB:
199  {
200  switch(pst->event)
201  {
202  case EVT_PHY_STUB_SLOT_IND:
203  {
204  fapi_slot_ind_t *slotIndMsg;
205 
206  CMCHKUNPK(oduUnpackPointer, (PTR *)&slotIndMsg, mBuf);
207  ODU_PUT_MSG_BUF(mBuf);
208 
209  procPhyMessages(slotIndMsg->header.msg_id, sizeof(fapi_slot_ind_t), (void*)slotIndMsg);
210  MAC_FREE_SHRABL_BUF(pst->region, pst->pool, slotIndMsg, sizeof(fapi_slot_ind_t));
211  break;
212  }
213 
214  case EVT_PHY_STUB_STOP_IND:
215  {
216  fapi_stop_ind_t *stopIndMsg;
217  CMCHKUNPK(oduUnpackPointer, (PTR *)&stopIndMsg, mBuf);
218  ODU_PUT_MSG_BUF(mBuf);
219 
220  procPhyMessages(stopIndMsg->header.msg_id, sizeof(fapi_stop_ind_t), (void*)stopIndMsg);
221  MAC_FREE_SHRABL_BUF(pst->region, pst->pool, stopIndMsg, sizeof(fapi_stop_ind_t));
222  break;
223  }
224  default:
225  {
226  DU_LOG("\nERROR --> LWR_MAC: Invalid event %d received from PHY STUB", pst->event);
227  }
228  }
229  break;
230  }
231 #endif
232 
233  default:
234  {
235  ODU_PUT_MSG_BUF(mBuf);
236  DU_LOG("\nERROR --> LWR MAC: Message from invalid source entity %d", pst->srcEnt);
237  ret = RFAILED;
238  }
239  }
240  return ret;
241 }
